Select Menu allows the users to select an option out of 25 options.

Lets Create Select Menu using DenoCord.

import { ActionRow, SelectMenu } from "https://deno.land/x/denocordts/mod.ts";

const row = new ActionRow().addComponents(
  new SelectMenu().setCustomId("testMenu").setOptions([
    {
      label: "Testing",
      value: "test",
      description: "An option for testing",
    },
  ])
);

await message.reply({ components: [row] });
